	<style>
	body {
	color:red;
	}
	</style> -->
	<!-- Using css - Type 3 -->
	<link href="styles.css" rel="stylesheet">
	<!-- Using css - Type 2
	<h1 style="color:red"> Welcome to Component Library </h1> -->
	
	Styles.css can be defined with tag level, class level, id level
	ol {
	color: black;
	}
	#header {
	color: cyan;
	}
	.medium-heading {

	}
